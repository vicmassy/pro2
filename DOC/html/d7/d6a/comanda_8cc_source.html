214  fixar_temps_rellotge();
215  else if (e[0].cos() == "sortir" and mida == 1)
216  e_acabament = true;
217  else if (e[0].es_consulta())
218  tractar_consulta();
219  else if (e[0].es_nombre()){

221  be = false;
222  else if (e[1].es_esborrat())
223  tractar_esborrat_parcial();
224  else
